import { Injectable } from '@angular/core';
import { HttpClient } from '@angular/common/http';
import { Subject, ReplaySubject } from 'rxjs';
import { IMyEventRoot } from './IMyEventRoot.interface';
import { Router } from '@angular/router';
import { IEvent } from './IEvent.interface';
import { IContactPerson } from './IContactPerson.interface';
import { IAddress } from './IAddress.interface';
import { AuthenticateService } from '../Auth/Authenticate.service';
import { map } from 'rxjs/operators';
import { stringify } from '@angular/core/src/render3/util';

@Injectable({
  providedIn: 'root',
})

export class EventService {
  currentRootEvent: IMyEventRoot;
  currentContactPerson: IContactPerson;
  currentEvent: IEvent;
  currentAddress: IAddress;
  maxEvents;
  editEventSubject = new ReplaySubject<IMyEventRoot>();
  eventsUpdatedSubject = new Subject<{ events: IMyEventRoot[], totalEvents: number}>();
  private events: IMyEventRoot[];
  reqEvent: any[] = [];
  private messageListener = new Subject<{message: string, isErrorMessage: boolean, fadingTime: number}>();
  totalEvents;
  private BACKEDEND_URL = 'https://newmousleheventappp.herokuapp.com';

  constructor(private http: HttpClient, private router: Router, private authService: AuthenticateService) { }


  getMessageListener() {
    return this.messageListener.asObservable();
}

  getEditEventListener() {
    return this.editEventSubject.asObservable();
  }

  eventsUpdatedSubjectListener() {
    return this.eventsUpdatedSubject.asObservable();
  }

  getLimitedAmountOfEvents(startAmount, endAmount) {
    const queryParams = `?startAmount=${startAmount}&endAmount=${endAmount}`;

    this.http.get<{ message: string, events: IMyEventRoot[], maxEvents: number }>
      ('http://localhost:3000/event' + queryParams)
      .subscribe(eventData => {
        this.events = eventData.events;
        this.eventsUpdatedSubject.next({
          events: [...this.events],
          totalEvents: this.events.length
        });

      });
  }

  getRequestedEvents(startAmount, endAmount) {
    const queryParams = `?startAmount=${startAmount}&endAmount=${endAmount}`;
    this.http.get<{ message: string, events: IMyEventRoot[], maxEvents: number }>
      ('http://localhost:3000/requestedEvents' + queryParams)
      .subscribe(eventData => {
        this.events = eventData.events;
        this.eventsUpdatedSubject.next({
          events: [...this.events],
          totalEvents: this.events.length
        });
      });

  }

  getRequstedThemeEvents(theme: string, eventsPerPage: number, currentPage: number) {
    const queryParams = `?pageSize=${eventsPerPage}&page=${currentPage}&theme=${theme}`;
    this.http
     .get<{ message: string, events: IMyEventRoot[], maxEvents: number }>(this.BACKEDEND_URL + '/theme' + queryParams)
     .subscribe(eventData => {
      eventData.events.filter(eventRoot => new Date(eventRoot.endTime) > new Date());
      this.events = eventData.events;
      this.eventsUpdatedSubject.next({
          events: [...this.events],
          totalEvents: this.events.length
        });
     });
  }


  declineRequstedEvent(event) {
    this.http.delete(this.BACKEDEND_URL + '/requestedEvents/' + event).subscribe(() => {
      this.eventsUpdatedSubject.next({
        events: [...this.events],
        totalEvents: this.events.length
      });
    });
  }

  postEvent(event: IEvent, address: IAddress, contactPerson: IContactPerson) {

    this.http.post<{message: string}>(this.BACKEDEND_URL + '/event'
      , this.createEventFormData(event, address, contactPerson)).subscribe(res => {

        this.messageListener.next(
          {
            message: res.message,
           isErrorMessage: false,
           fadingTime: 6000
          });

      }, error => {

        this.messageListener.next(
          {
          message: 'Noget gik galt prøv at trykke F5 eller opdater siden',
           isErrorMessage: true,
           fadingTime: 6000
          });
      });
  }

  getUserEvents(startAmount, endAmount) {
    const queryParams = `?startAmount=${startAmount}&endAmount=${endAmount}`;
    this.http.get<{ message: string, events: IMyEventRoot[], maxEvents: number }>('http://localhost:3000/event/user/events' + queryParams)
    .subscribe(eventData => {
      this.events = eventData.events;
      this.eventsUpdatedSubject.next({
        events: [...this.events],
        totalEvents: this.events.length
      });
    });
  }


  publishEvent(event: IMyEventRoot) {

    this.http.post<{message: string}>(this.BACKEDEND_URL + '/event/publish', event).subscribe(res => {

      this.messageListener.next(
        {
          message: res.message,
         isErrorMessage: false,
         fadingTime: 1000
        });

    }, error => {
      this.messageListener.next(
        {
        message: 'Noget gik galt prøv at trykke F5 eller opdater siden',
         isErrorMessage: true,
         fadingTime: 1000
        });
    }
    );
  }

  deleteEvent(id: string) {
    return this.http.delete<{message: string}>(this.BACKEDEND_URL + '/event/' + id)
    .subscribe(res => {
      this.messageListener.next(
        {
         message: res.message,
         isErrorMessage: true,
         fadingTime: 3000
        });
   }, error => {
      this.messageListener.next(
        {
          message: error.error.message,
         isErrorMessage: false,
         fadingTime: 3000
        });

    });

  }

  updateEvent(id: string, event: IEvent, address: IAddress, contactPerson: IContactPerson) {

    this.http.put<{message: string}>(this.BACKEDEND_URL + '/event/' + id, this.createEventFormData(event, address, contactPerson))
      .subscribe(res => {
        console.log(res);

        this.messageListener.next(
        {
          message: res.message,
         isErrorMessage: false,
         fadingTime: 3000
        });

    }, error => {
      this.messageListener.next(
        {
        message: error.error.message,
         isErrorMessage: true,
         fadingTime: 3000
        });

      });
  }

  getOneEvent(_id: string) {
    this.http.get<IMyEventRoot>(this.BACKEDEND_URL + '/event/' + _id)
      .subscribe(event => {
        this.currentRootEvent = event;
        this.editEventSubject.next(event);
      });
  }

  private createEventFormData(event: IEvent, address: IAddress, contactPerson: IContactPerson) {

    if ((event || address || contactPerson ) === undefined) {
      return;
    }
    if (typeof event.imagePath === 'object') {

      const fd = new FormData();
      fd.append('name', event.name);

      fd.append('description', event.description);

      fd.append('theme', event.theme);

      fd.append('ageLimit', event.ageLimit.toString());

      fd.append('startTime', event.startTime);

      fd.append('endTime', event.endTime);

      fd.append('cost', event.cost.toString());

      fd.append('imagePath', event.imagePath, event.imagePath.name);

      fd.append('line', address.line);

      fd.append('zipCode', address.zipCode.toString());

      fd.append('city', address.city);

      fd.append('contactName', contactPerson.contactName);

      fd.append('contactPhone', contactPerson.contactPhone);

      fd.append('contactEmail', contactPerson.contactEmail);

      fd.append('creatorId', this.authService.getUserId());
      return fd;
    } else {
      const fd = new FormData();

      fd.append('name', event.name);

      fd.append('description', event.description);

      fd.append('theme', event.theme);

      fd.append('ageLimit', event.ageLimit.toString());

      fd.append('startTime', event.startTime);

      fd.append('endTime', event.endTime);

      fd.append('cost', event.cost.toString());

      fd.append('imagePath', event.imagePath);

      fd.append('line', address.line);

      fd.append('zipCode', address.zipCode.toString());

      fd.append('city', address.city);

      fd.append('contactName', contactPerson.contactName);

      fd.append('contactPhone', contactPerson.contactPhone);

      fd.append('contactEmail', contactPerson.contactEmail);

      fd.append('creatorId',  this.authService.getUserId());
      return fd;
    }
  }

  sendDeclinedMail(eventName, personName, email, reason) {
    const declinedMailObject = {
      eventName,
      personName,
      email,
      reason
    };

    this.http.post('http://localhost:3000/user/declinedMail/', declinedMailObject).subscribe(res => console.log(res));
  }

  sendPublishedMail(eventName, personName, email) {
    const declinedMailObject = {
      eventName,
      personName,
      email
    };

    console.log(declinedMailObject);
    this.http.post('http://localhost:3000/user/publishedMail/', declinedMailObject).subscribe(res => console.log(res));
  }
}
